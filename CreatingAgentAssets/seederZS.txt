-- ============================================================
-- 5. VERIFICACION
-- ============================================================
SELECT z.slug AS zona, c.name AS comunidad, c.safety_status, c.role_required
FROM communities c
JOIN zones z ON z.id = c.zone_id
WHERE z.slug = 'parque-metropolitano'
ORDER BY c.name;

SELECT n.name, n.type, n.status, n.pos_x, n.pos_y, COALESCE(n.lorawan_eui, 'simulado') AS eui
FROM nodes n
JOIN zones z ON z.id = n.zone_id
WHERE z.slug = 'parque-metropolitano'
ORDER BY n.name;
