-- ============================================================
-- 4. VERIFICACIÓN FINAL
-- ============================================================

SELECT
  z.name                                    AS zona,
  n.name                                    AS nodo,
  n.type                                    AS tipo,
  n.status                                  AS estado,
  COALESCE(n.lorawan_eui, '— simulado —')  AS eui,
  n.x                                       AS "pos_x%",
  n.y                                       AS "pos_y%"
FROM nodes n
JOIN zones z ON z.id = n.zone_id
ORDER BY z.name, n.type DESC, n.name;
