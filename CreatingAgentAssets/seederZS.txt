-- 3. VERIFICACIÓN INMEDIATA
-- ============================================================
SELECT
  n.name                        AS nodo,
  n.type                        AS tipo,
  n.status                      AS estado,
  COALESCE(n.lorawan_eui, '— simulado —') AS eui,
  ROUND(n.pos_x::numeric, 1)    AS "pos_x%",
  ROUND(n.pos_y::numeric, 1)    AS "pos_y%",
  n.lat,
  n.lng
FROM nodes n
JOIN zones z ON z.id = n.zone_id
WHERE z.slug = 'akron'
ORDER BY n.type DESC, n.name;
