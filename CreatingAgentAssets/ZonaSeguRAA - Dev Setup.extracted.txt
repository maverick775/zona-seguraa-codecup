1. Decisión: Vercel sobre AWS
TL;DR: Usar Vercel + Supabase. No usar AWS EC2 ni Lambda para el MVP del hackathon.

Justificación estratégica: En un hackathon, cada hora de configuración de infraestructura es una hora menos de producto. Vercel + Supabase permiten estar en producción en menos de 10 minutos. AWS es la plataforma correcta para producción real — está contemplada en el roadmap post-CodeCup para alta disponibilidad y escala [file:2].

4. Hook de Realtime — Plantilla
// hooks/useRealtimeAlerts.js
import { useEffect, useState } from 'react'
import { createClient } from '@/lib/supabase'
export function useRealtimeAlerts(zoneId) {
const [alerts, setAlerts] = useState([])
const supabase = createClient()
useEffect(() => {
// Carga inicial
supabase
.from('alerts')
.select('*, nodes(*), alert_votes(*)')
.eq('zone_id', zoneId)
.order('created_at', { ascending: false })
.then(({ data }) => setAlerts(data || []))
// Suscripción a cambios en tiempo real
const channel = supabase
.channel(`alerts:zone:${zoneId}`)
.on('postgres_changes', {
event: '*', // INSERT, UPDATE, DELETE
schema: 'public',
table: 'alerts',
filter: `zone_id=eq.${zoneId}`
}, (payload) => {
setAlerts(prev => {
if (payload.eventType === 'INSERT') return [payload.new, ...prev]
if (payload.eventType === 'UPDATE') return prev.map(a => a.id === payload.new.id ? payload.new : a)
return prev
})
})
.subscribe()
return () => supabase.removeChannel(channel)
}, [zoneId])
return alerts
}

5.2 Estructura del payload que envía Loriot
{
"EUI": "0004A30B001B9B27",
"ts": 1742395200000,
"rssi": -87,
"snr": 8.2,
"fcnt": 42,
"port": 1,
"data": "0164"
}
data es hex: 01 = estado (1=ALERTA), 64 = batería (100%)
5.3 Implementación del webhook
// app/api/iot/uplink/route.js
import { createServerClient } from '@/lib/supabase-server'
import { NextResponse } from 'next/server'
export async function POST(request) {
const supabase = createServerClient()
const body = await request.json()
const eui = body.EUI
const hex = body.data || ''
const alertState = parseInt(hex.substring(0, 2), 16) // Byte 0
const batteryPct = parseInt(hex.substring(2, 4), 16) // Byte 1
// Registrar uplink en log
await supabase.from('iot_uplinks').insert({
node_eui: eui,
payload_hex: hex,
alert_state: alertState,
battery_pct: batteryPct
})
// Buscar el nodo correspondiente
const { data: node } = await supabase
.from('nodes')
.select('*')
.eq('lorawan_eui', eui)
.single()
if (!node) return NextResponse.json({ error: 'Node not found' }, { status: 404 })
// Actualizar estado del nodo
const nodeStatus = alertState > 0 ? 'alert' : 'online'
await supabase.from('nodes').update({ status: nodeStatus, last_seen_at: new Date().toISOString() }).eq('id', node.id)
// Si hay alerta activa, crear o actualizar alerta en la zona
if (alertState > 0) {
const { data: existingAlert } = await supabase
.from('alerts')
.select('id')
.eq('node_id', node.id)
.in('status', ['active', 'in_progress'])
.single()
if (!existingAlert) {
await supabase.from('alerts').insert({
zone_id: node.zone_id,
node_id: node.id,
type: 'security',
level: alertState,
status: 'active',
description: `Alerta activada por nodo AVP3-ALFA (${eui})`,
language: 'es',
created_by_nick: 'AVP3-SYSTEM'
})
}
}
return NextResponse.json({ ok: true })
}
6. Script Wizard of Oz
// scripts/demo-mode.js
// Ejecutar con: node scripts/demo-mode.js
const BASE_URL = process.env.DEMO_URL || 'https://zonaseguraa.com'
const ZONE_ID = process.env.ZONE_ID || 'AKRON_ZONE_UUID'
const DELAY = parseInt(process.env.DELAY) || 5000 // ms entre pasos
async function post(path, body) {
const res = await fetch(`${BASE_URL}${path}`, {
method: 'POST',
headers: { 'Content-Type': 'application/json' },
body: JSON.stringify(body)
})
return res.json()
}
async function patch(path, body) {
const res = await fetch(`${BASE_URL}${path}`, {
method: 'PATCH',
headers: { 'Content-Type': 'application/json' },
body: JSON.stringify(body)
})
return res.json()
}
async function sleep(ms) { return new Promise(r => setTimeout(r, ms)) }

7. Configuración del Mapa Leaflet — Estadio Akron
// components/StadiumMap.js
'use client'
import { useEffect, useRef } from 'react'
import L from 'leaflet'
import 'leaflet/dist/leaflet.css'
// Coordenadas reales del Estadio Akron, Guadalajara
const AKRON_CENTER = [20.6892, -103.4653]
const AKRON_BOUNDS = [
[20.6870, -103.4680], // SW
[20.6915, -103.4625] // NE
]
export default function StadiumMap({ nodes, alerts, onNodeClick }) {
const mapRef = useRef(null)
const mapInstance = useRef(null)
useEffect(() => {
if (mapInstance.current) return
mapInstance.current = L.map(mapRef.current, {
center: AKRON_CENTER,
zoom: 17,
zoomControl: true
})
// Base layer: OpenStreetMap
L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
attribution: '© OpenStreetMap contributors'
}).addTo(mapInstance.current)
// Overlay: imagen del estadio (agregar cuando se tenga el archivo)
// L.imageOverlay('/estadio-akron.jpg', AKRON_BOUNDS).addTo(mapInstance.current)
}, [])
useEffect(() => {
if (!mapInstance.current) return
// Actualizar marcadores cuando cambien nodos
nodes?.forEach(node => {
const color = node.status === 'alert' ? 'red' : node.status === 'online' ? 'green' : 'gray'
const icon = L.divIcon({
className: '',
html: `<div style="background:${color};width:14px;height:14px;border-radius:50%;border:2px solid white;"></div>`
})
L.marker([node.lat || AKRON_CENTER[0], node.lng || AKRON_CENTER[1]], { icon })
.addTo(mapInstance.current)
.bindPopup(`<b>${node.name}</b><br>Estado: ${node.status}`)
})
}, [nodes])
return <div ref={mapRef} style={{ height: '400px', width: '100%', borderRadius: '8px' }} />
}
